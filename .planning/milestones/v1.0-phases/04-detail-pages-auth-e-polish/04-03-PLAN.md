```css
@media print {
  /* Hide application chrome */
  [data-sidebar],
  nav:not([role="tablist"]),
  header,
  .no-print,
  button {
    display: none !important;
  }

  /* Full width for content area */
  main {
    padding: 0 !important;
    overflow: visible !important;
  }

  /* Remove flex layout constraints */
  .flex.h-screen {
    display: block !important;
    height: auto !important;
  }

  /* Show all tab sections in print (not just active tab) */
  [data-print-section] {
    display: block !important;
  }

  /* Prevent page breaks inside cards */
  [class*="rounded-"] {
    break-inside: avoid;
  }

  /* Remove decorative elements that waste ink */
  * {
    box-shadow: none !important;
  }

  /* Ensure text is black for readability */
  body {
    color: #000 !important;
    background: white !important;
  }
}
```
