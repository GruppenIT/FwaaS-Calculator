**Part A — Add getFatalDeadlineSummary to api.ts:**

1. Add the interface and function near the existing prazos section (after `listarPrazos`):

```typescript
export interface FatalDeadlineSummary {
  today: number;
  tomorrow: number;
}

export async function getFatalDeadlineSummary(): Promise<FatalDeadlineSummary> {
  const prazos = await listarPrazos({ status: 'pendente' });
  const hoje = new Date();
  hoje.setHours(0, 0, 0, 0);
  const amanha = new Date(hoje);
  amanha.setDate(amanha.getDate() + 1);

  let today = 0;
  let tomorrow = 0;
  for (const p of prazos) {
    if (p.suspenso) continue;
    const d = new Date(p.dataFatal);
    d.setHours(0, 0, 0, 0);
    const diffMs = d.getTime() - hoje.getTime();
    const diffDays = Math.round(diffMs / (1000 * 60 * 60 * 24));
    if (diffDays === 0) today++;
    else if (diffDays === 1) tomorrow++;
  }
  return { today, tomorrow };
}
```

Key logic: filter only status=pendente via API param, then client-side filter out suspenso=true, then compute date diff with midnight normalization. Count prazos where dataFatal is today (diffDays=0) or tomorrow (diffDays=1).

**Part B — Create use-fatal-deadlines.ts hook:**

Create new file `packages/app-desktop/src/hooks/use-fatal-deadlines.ts`:

```typescript
import { useState, useEffect } from 'react';
import { getFatalDeadlineSummary, type FatalDeadlineSummary } from '../lib/api';

const POLL_INTERVAL = 5 * 60 * 1000; // 5 minutes

export function useFatalDeadlines(): FatalDeadlineSummary {
  const [summary, setSummary] = useState<FatalDeadlineSummary>({ today: 0, tomorrow: 0 });

  useEffect(() => {
    let cancelled = false;

    async function fetchSummary() {
      try {
        const data = await getFatalDeadlineSummary();
        if (!cancelled) setSummary(data);
      } catch {
        // Silent failure — banner polling must never crash AppLayout
      }
    }

    fetchSummary();
    const interval = setInterval(fetchSummary, POLL_INTERVAL);

    return () => {
      cancelled = true;
      clearInterval(interval);
    };
  }, []);

  return summary;
}
```

The hook: (1) fetches on mount with initial state { today: 0, tomorrow: 0 } so banner is hidden until data arrives, (2) polls every 5 min, (3) silently swallows errors, (4) cleans up interval and cancelled flag on unmount for React strict mode safety.

**Part A — Create deadline-banner.tsx:**

Create new file `packages/app-desktop/src/components/layout/deadline-banner.tsx`:

```tsx
import { useNavigate } from 'react-router-dom';
import { TriangleAlert } from 'lucide-react';
import { useFatalDeadlines } from '../../hooks/use-fatal-deadlines';

export function DeadlineBanner() {
  const { today, tomorrow } = useFatalDeadlines();
  const navigate = useNavigate();

  if (today === 0 && tomorrow === 0) return null;

  return (
    <div
      role="alert"
      onClick={() => navigate('/app/prazos')}
      className="mx-6 mt-4 mb-0 rounded-[var(--radius-md)] border border-causa-tier-fatal/30 bg-causa-tier-fatal/8 p-4 cursor-pointer hover:bg-causa-tier-fatal/12 transition-causa"
    >
      <div className="flex items-start gap-3">
        <TriangleAlert size={20} className="text-causa-tier-fatal mt-0.5 shrink-0" />
        <div className="flex-1 min-w-0">
          {today > 0 && (
            <p className="text-sm-causa font-medium text-causa-tier-fatal">
              {today} {today === 1 ? 'prazo fatal vence' : 'prazos fatais vencem'} hoje
            </p>
          )}
          {tomorrow > 0 && (
            <p className={`text-sm-causa font-medium text-causa-tier-fatal ${today > 0 ? 'mt-0.5' : ''}`}>
              {tomorrow} {tomorrow === 1 ? 'prazo fatal vence' : 'prazos fatais vencem'} amanha
            </p>
          )}
        </div>
      </div>
    </div>
  );
}
```

Key design decisions:
- `role="alert"` for accessibility (screen readers announce it)
- Non-dismissible: no X button, no close mechanism
- Clickable entire banner navigates to `/app/prazos`
- Uses `causa-tier-fatal` Tailwind alias which maps to `var(--color-tier-fatal)` = `#dc2626`
- Opacity modifiers: `/8` bg, `/30` border, `/12` hover bg — follows UpdateBanner pattern
- Differentiates singular/plural: "prazo fatal vence" vs "prazos fatais vencem"
- When both today and tomorrow exist, both lines show with small margin between them
- `transition-causa` for smooth hover effect
- Returns null when both counts are 0 (no render, no DOM element)
